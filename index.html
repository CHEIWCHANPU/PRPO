🔗 การเชื่อม Google Apps Script

ทำตามขั้นตอนด้านล่างเพื่อเชื่อม Google Sheet เข้ากับ Dashboard

1เปิด Google Sheet → Extensions → Apps Script
2ลบโค้ดเดิมทั้งหมด แล้ววางโค้ดด้านล่าง
3Save 💾 แล้ว Deploy → New Deployment
4Type = Web App, Execute as = Me, Access = Anyone → Deploy
5คัดลอก URL มาวางในช่อง "Web App URL" ด้านล่าง แล้วกด บันทึก

📋 Apps Script Code

// ─── CHEIW CHAN PR/PO TRACKER · Apps Script ─── const SHEET_NAME = "PR/PO"; const SS = SpreadsheetApp.getActiveSpreadsheet(); function doGet(e) { const action = e.parameter.action || "getData"; let result; try { if (action === "getData") result = getData(); else if (action === "add") result = addRow(e.parameter); else if (action === "edit") result = editRow(e.parameter); else if (action === "del") result = deleteRow(e.parameter); else result = { error: "Unknown action" }; } catch(err) { result = { error: err.message }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function getData() { const sheet = SS.getSheetByName(SHEET_NAME); const rows = sheet.getDataRange().getValues(); const data = []; const today = new Date(); today.setHours(0,0,0,0); for (let i = 8; i < rows.length; i++) { const r = rows[i]; if (!r[0] && !r[1]) continue; // --- แปลงสถานะ: รองรับ checkbox (TRUE/FALSE) และ text --- let statusRaw = r[9]; let status; if (statusRaw === true || String(statusRaw).toUpperCase() === "TRUE") status = "เสร็จแล้ว ✅"; else if (statusRaw === false || String(statusRaw).toUpperCase() === "FALSE") status = "กำลังดำเนินการ ♻"; else status = String(statusRaw || "กำลังดำเนินการ ♻"); // --- คำนวณ daysLeft ใหม่เสมอ (ไม่อ่านจาก Sheet เพื่อกัน formula เพี้ยน) --- let daysLeft; if (status.includes("เสร็จแล้ว")) { daysLeft = "Done"; } else { const due = r[6]; if (due instanceof Date) { const d = new Date(due); d.setHours(0,0,0,0); daysLeft = Math.round((d - today) / 86400000); } else { daysLeft = ""; } } data.push({ no: r[0], partNo: r[1], partName: r[2], model: r[3], volume: r[4], prDate: formatDate(r[5]), dueDate: formatDate(r[6]), actualDate: formatDate(r[7]), location: r[8], status: status, maker: r[10], daysLeft: daysLeft, poNo: r[12], note: r[13], rowIndex: i + 1 }); } return { success: true, data, updated: new Date().toLocaleString("th-TH") }; } function formatDate(v) { if (!v) return ""; if (v instanceof Date) return Utilities.formatDate(v, "Asia/Bangkok", "d/M/yyyy"); return String(v); } function addRow(p) { const sheet = SS.getSheetByName(SHEET_NAME); const last = sheet.getLastRow(); const no = last - 7; sheet.appendRow([ no, p.partNo, p.partName, p.model, Number(p.volume) || 0, p.prDate, p.dueDate, p.actualDate || "", p.location, p.status, p.maker, "", p.poNo || "", p.note || "" ]); updateTimestamp(); return { success: true }; } function editRow(p) { const sheet = SS.getSheetByName(SHEET_NAME); const ri = parseInt(p.rowIndex); // ถ้า column สถานะใช้ checkbox ให้แปลงกลับเป็น TRUE/FALSE // ถ้าใช้ text ให้เขียน text ตรง ๆ — ตรวจสอบจาก cell เดิม const existingCell = sheet.getRange(ri, 10); const existingVal = existingCell.getValue(); let statusToWrite = p.status; if (typeof existingVal === "boolean") { // cell เป็น checkbox → เขียนกลับเป็น boolean statusToWrite = p.status.includes("เสร็จแล้ว"); } const cols = [ p.partNo, p.partName, p.model, Number(p.volume) || 0, p.prDate, p.dueDate, p.actualDate || "", p.location, statusToWrite, p.maker ]; for (let c = 0; c < cols.length; c++) { sheet.getRange(ri, c + 2).setValue(cols[c]); } // col L (index 12) = daysLeft — ข้ามไม่เขียนทับ (ปล่อยให้ formula Sheet คำนวณเอง) sheet.getRange(ri, 13).setValue(p.poNo || ""); sheet.getRange(ri, 14).setValue(p.note || ""); updateTimestamp(); return { success: true }; } function deleteRow(p) { const sheet = SS.getSheetByName(SHEET_NAME); sheet.deleteRow(parseInt(p.rowIndex)); updateTimestamp(); return { success: true }; } function updateTimestamp() { const sheet = SS.getSheetByName(SHEET_NAME); sheet.getRange("A22").setValue( "อัปเดตล่าสุด: " + Utilities.formatDate(new Date(), "Asia/Bangkok", "dd/MM/yyyy HH:mm") ); }
