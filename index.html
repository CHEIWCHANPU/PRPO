🔗 การเชื่อม Google Apps Script กับ Dashboard นี้

ทำตามขั้นตอนด้านล่างเพื่อเชื่อม Google Sheet ของคุณเข้ากับ Dashboard

1เปิด Google Sheet ของคุณ → คลิก Extensions → Apps Script
2ลบโค้ดเดิมทิ้งทั้งหมด แล้ว วางโค้ด Apps Script ด้านล่างนี้
3คลิก Save 💾 แล้วคลิก Deploy → New Deployment
4เลือก Type = Web App, Execute as = Me, Access = Anyone → Deploy
5คัดลอก URL ที่ได้ มาวางในช่อง "Web App URL" ด้านล่าง แล้วกด บันทึก
6อัปโหลดไฟล์ index.html ขึ้น GitHub Pages แล้วเปิด URL จาก GitHub

📋 Apps Script Code (วาง ทั้งหมดใน Code.gs)

// ─── CHEIW CHAN PR/PO TRACKER · Apps Script ─── const SHEET_NAME = "PR/PO"; const SS = SpreadsheetApp.getActiveSpreadsheet(); function doGet(e) { const action = e.parameter.action || "getData"; let result; try { if (action === "getData") result = getData(); else if (action === "add") result = addRow(e.parameter); else if (action === "edit") result = editRow(e.parameter); else if (action === "del") result = deleteRow(e.parameter); else result = { error: "Unknown action" }; } catch(err) { result = { error: err.message }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function getData() { const sheet = SS.getSheetByName(SHEET_NAME); const rows = sheet.getDataRange().getValues(); const data = []; for (let i = 8; i < rows.length; i++) { const r = rows[i]; if (!r[0] && !r[1]) continue; data.push({ no: r[0], partNo: r[1], partName: r[2], model: r[3], volume: r[4], prDate: formatDate(r[5]), dueDate: formatDate(r[6]), actualDate: formatDate(r[7]), location: r[8], status: r[9], maker: r[10], daysLeft: r[11], poNo: r[12], note: r[13], rowIndex: i + 1 }); } return { success: true, data, updated: new Date().toLocaleString("th-TH") }; } function formatDate(v) { if (!v) return ""; if (v instanceof Date) return Utilities.formatDate(v, "Asia/Bangkok", "dd/MM/yyyy"); return String(v); } function addRow(p) { const sheet = SS.getSheetByName(SHEET_NAME); const last = sheet.getLastRow(); const no = last - 7; sheet.appendRow([ no, p.partNo, p.partName, p.model, Number(p.volume) || 0, p.prDate, p.dueDate, p.actualDate || "", p.location, p.status, p.maker, "", p.poNo || "", p.note || "" ]); updateTimestamp(); return { success: true }; } function editRow(p) { const sheet = SS.getSheetByName(SHEET_NAME); const ri = parseInt(p.rowIndex); // col B(2)=partNo, C(3)=partName, D(4)=model, E(5)=volume, // F(6)=prDate, G(7)=dueDate, H(8)=actualDate, // I(9)=location, J(10)=status, K(11)=maker, // col L(12)=daysLeft ข้ามไม่เขียนทับ (อาจเป็น formula หรือ checkbox), // M(13)=poNo, N(14)=note const colsBeforeDaysLeft = [ p.partNo, p.partName, p.model, Number(p.volume) || 0, p.prDate, p.dueDate, p.actualDate || "", p.location, p.status, p.maker ]; for (let c = 0; c < colsBeforeDaysLeft.length; c++) { sheet.getRange(ri, c + 2).setValue(colsBeforeDaysLeft[c]); } // ข้าม col L (daysLeft) ไม่เขียนทับ sheet.getRange(ri, 13).setValue(p.poNo || ""); // col M sheet.getRange(ri, 14).setValue(p.note || ""); // col N updateTimestamp(); return { success: true }; } function deleteRow(p) { const sheet = SS.getSheetByName(SHEET_NAME); sheet.deleteRow(parseInt(p.rowIndex)); updateTimestamp(); return { success: true }; } function updateTimestamp() { const sheet = SS.getSheetByName(SHEET_NAME); sheet.getRange("A22").setValue( "อัปเดตล่าสุด: " + Utilities.formatDate(new Date(), "Asia/Bangkok", "dd/MM/yyyy HH:mm") + " | จัดทำโดย: CHEIW CHAN DATA" ); }
