🔗 การเชื่อม Google Apps Script

ทำตามขั้นตอนด้านล่างเพื่อเชื่อม Google Sheet เข้ากับ Dashboard แบบ Real-time

1เปิด Google Sheet (Sheet: 📋 กรอกข้อมูล) → Extensions → Apps Script
2ลบโค้ดเดิมทั้งหมด แล้ววางโค้ดด้านล่าง
3กด Save 💾 แล้วเลือก Deploy → New Deployment
4เลือก Type = Web App, Execute as = Me, Access = Anyone → Deploy
5คัดลอก URL ที่ได้มาวางในช่อง Web App URL ด้านล่าง แล้วกด บันทึก

📋 Apps Script Code

// ═══════════════════════════════════════════════════════════════════════════ // CHIEWCHAN INDUSTRY CO.,LTD. — PR/PO TRACKER · Apps Script v4 // Sheet "📋 กรอกข้อมูล": Header Row5 | Data Row6+ // Sheet "supply": Header Row2 | Data Row3+ // // PR/PO Columns (A=1): // A=NO. | B=PART NO. | C=PART NAME | D=MODEL | E=VOLUME // F=เปิด P/R | G=วันนัดส่ง | H=วันเข้าจริง | I=สถานที่ // J=สถานะ | K=MAKER | L=เหลือ(วัน) | M=PO/ORDER | N=หมายเหตุ // O=STARRED (TRUE/FALSE) | P=COMMENTS (JSON string) ← ใหม่ v4 // // Supply Columns (A=1): // A=NO. | B=ชื่อบริษัท | C=ชื่อผู้ติดต่อ | D=เบอร์โทร // E=EMAIL | F=หมวดหมู่สินค้า | G=LEADTIME | H=หมายเหตุ // ═══════════════════════════════════════════════════════════════════════════ const SHEET_NAME = "📋 กรอกข้อมูล"; const SUPPLY_SHEET = "supply"; const DATA_START_ROW = 6; const SUPPLY_START_ROW = 3; const HEADER_ROW = 5; const TOTAL_COLS = 16; // A–P (เพิ่ม O=STARRED, P=COMMENTS) function doGet(e) { const output = ContentService.createTextOutput(); output.setMimeType(ContentService.MimeType.JSON); const action = (e && e.parameter && e.parameter.action) || "getData"; let result; try { if (action === "getData") result = getData(); else if (action === "add") result = addRow(e.parameter); else if (action === "edit") result = editRow(e.parameter); else if (action === "del") result = deleteRow(e.parameter); else if (action === "setStar") result = setStar(e.parameter); else if (action === "setComments") result = setComments(e.parameter); else if (action === "getSuppliers") result = getSuppliers(); else if (action === "addSupplier") result = addSupplier(e.parameter); else if (action === "editSupplier") result = editSupplier(e.parameter); else if (action === "delSupplier") result = delSupplier(e.parameter); else if (action === "ping") result = { success: true, message: "pong" }; else result = { success: false, error: "Unknown action: " + action }; } catch (err) { result = { success: false, error: err.message, stack: err.stack }; } output.setContent(JSON.stringify(result)); return output; } // ─── PR/PO: READ ──────────────────────────────────────────────────────── function getData() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return { success: false, error: "ไม่พบ Sheet: " + SHEET_NAME }; ensureHeaders(sheet); const lastRow = sheet.getLastRow(); if (lastRow < DATA_START_ROW) return { success: true, data: [], updated: fmtNow() }; const numRows = lastRow - DATA_START_ROW + 1; const rows = sheet.getRange(DATA_START_ROW, 1, numRows, TOTAL_COLS).getValues(); const today = new Date(); today.setHours(0, 0, 0, 0); const data = []; rows.forEach((r, idx) => { if (!r[1] && !r[2]) return; const status = normalizeStatus(r[9]); let daysLeft; if (status.includes("เสร็จแล้ว")) { daysLeft = "Done"; } else { const due = r[6]; if (due instanceof Date && !isNaN(due)) { const d = new Date(due); d.setHours(0, 0, 0, 0); daysLeft = Math.round((d - today) / 86400000); } else { daysLeft = ""; } } // คอลัมน์ O (index 14) = STARRED, P (index 15) = COMMENTS let comments = []; try { comments = JSON.parse(String(r[15] || "[]")); } catch(e) { comments = []; } data.push({ no: r[0] || (idx + 1), partNo: String(r[1] || "").trim(), partName: String(r[2] || "").trim(), model: String(r[3] || "").trim(), volume: r[4] || 0, prDate: fmtDate(r[5]), dueDate: fmtDate(r[6]), actualDate: fmtDate(r[7]), location: String(r[8] || "").trim(), status: status, maker: String(r[10] || "").trim(), daysLeft: daysLeft, poNo: String(r[12] || "").trim(), note: String(r[13] || "").trim(), starred: r[14] === true || String(r[14]).toUpperCase() === "TRUE", comments: comments, rowIndex: DATA_START_ROW + idx }); }); return { success: true, data: data, updated: fmtNow() }; } // ─── PR/PO: ADD ───────────────────────────────────────────────────────── function addRow(p) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return { success: false, error: "ไม่พบ Sheet: " + SHEET_NAME }; ensureHeaders(sheet); const lastRow = sheet.getLastRow(); const no = lastRow >= DATA_START_ROW ? lastRow - DATA_START_ROW + 2 : 1; sheet.appendRow([ no, p.partNo||"", p.partName||"", p.model||"", Number(p.volume)||0, p.prDate||"", p.dueDate||"", p.actualDate||"", p.location||"", p.status||"กำลังดำเนินการ ♻", p.maker||"", "", p.poNo||"", p.note||"", false, "[]" ]); updateTimestamp(sheet); return { success: true }; } // ─── PR/PO: EDIT ──────────────────────────────────────────────────────── function editRow(p) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return { success: false, error: "ไม่พบ Sheet: " + SHEET_NAME }; const ri = parseInt(p.rowIndex); if (isNaN(ri) || ri < DATA_START_ROW) return { success: false, error: "rowIndex ไม่ถูกต้อง" }; const existingVal = sheet.getRange(ri, 10).getValue(); let statusToWrite = p.status || "กำลังดำเนินการ ♻"; if (typeof existingVal === "boolean") statusToWrite = p.status && p.status.includes("เสร็จแล้ว"); const cols = [ p.partNo||"", p.partName||"", p.model||"", Number(p.volume)||0, p.prDate||"", p.dueDate||"", p.actualDate||"", p.location||"", statusToWrite, p.maker||"" ]; for (let c = 0; c < cols.length; c++) sheet.getRange(ri, c + 2).setValue(cols[c]); sheet.getRange(ri, 13).setValue(p.poNo||""); sheet.getRange(ri, 14).setValue(p.note||""); // ไม่แตะ col O (STARRED) และ P (COMMENTS) เมื่อ edit ข้อมูลหลัก updateTimestamp(sheet); return { success: true }; } // ─── PR/PO: DELETE ────────────────────────────────────────────────────── function deleteRow(p) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return { success: false, error: "ไม่พบ Sheet: " + SHEET_NAME }; const ri = parseInt(p.rowIndex); if (isNaN(ri) || ri < DATA_START_ROW) return { success: false, error: "rowIndex ไม่ถูกต้อง" }; sheet.deleteRow(ri); renumberColumn(sheet); updateTimestamp(sheet); return { success: true }; } // ─── PR/PO: SET STAR ──────────────────────────────────────────────────── function setStar(p) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return { success: false, error: "ไม่พบ Sheet: " + SHEET_NAME }; const ri = parseInt(p.rowIndex); if (isNaN(ri) || ri < DATA_START_ROW) return { success: false, error: "rowIndex ไม่ถูกต้อง" }; const val = p.starred === "true" || p.starred === true; sheet.getRange(ri, 15).setValue(val); // คอลัมน์ O return { success: true }; } // ─── PR/PO: SET COMMENTS ──────────────────────────────────────────────── function setComments(p) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return { success: false, error: "ไม่พบ Sheet: " + SHEET_NAME }; const ri = parseInt(p.rowIndex); if (isNaN(ri) || ri < DATA_START_ROW) return { success: false, error: "rowIndex ไม่ถูกต้อง" }; // p.comments คือ JSON string ของ array const raw = p.comments || "[]"; try { JSON.parse(raw); } catch(e) { return { success: false, error: "comments ไม่ใช่ JSON ที่ถูกต้อง" }; } sheet.getRange(ri, 16).setValue(raw); // คอลัมน์ P return { success: true }; } // ─── SUPPLIER: READ ───────────────────────────────────────────────────── function getSuppliers() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SUPPLY_SHEET); if (!sheet) return { success: false, error: "ไม่พบ Sheet: " + SUPPLY_SHEET }; const lastRow = sheet.getLastRow(); if (lastRow < SUPPLY_START_ROW) return { success: true, data: [] }; const numRows = lastRow - SUPPLY_START_ROW + 1; const rows = sheet.getRange(SUPPLY_START_ROW, 1, numRows, 8).getValues(); const data = []; rows.forEach((r, idx) => { if (!r[1]) return; data.push({ id: SUPPLY_START_ROW + idx, no: r[0] || (idx + 1), name: String(r[1] || "").trim(), contact: String(r[2] || "").trim(), phone: String(r[3] || "").trim(), email: String(r[4] || "").trim(), category: String(r[5] || "").trim(), leadtime: r[6] || "", note: String(r[7] || "").trim(), rowIndex: SUPPLY_START_ROW + idx }); }); return { success: true, data: data }; } // ─── SUPPLIER: ADD ────────────────────────────────────────────────────── function addSupplier(p) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SUPPLY_SHEET); if (!sheet) return { success: false, error: "ไม่พบ Sheet: " + SUPPLY_SHEET }; const maxRow = Math.max(sheet.getLastRow(), SUPPLY_START_ROW - 1); let targetRow = -1; if (maxRow >= SUPPLY_START_ROW) { const vals = sheet.getRange(SUPPLY_START_ROW, 2, maxRow - SUPPLY_START_ROW + 1, 1).getValues(); for (let i = 0; i < vals.length; i++) { if (!vals[i][0]) { targetRow = SUPPLY_START_ROW + i; break; } } } if (targetRow === -1) targetRow = maxRow + 1; sheet.getRange(targetRow, 1, 1, 8).setValues([[ 0, p.name||"", p.contact||"", p.phone||"", p.email||"", p.category||"", Number(p.leadtime)||"", p.note||"" ]]); renumberSupply(sheet); return { success: true }; } // ─── SUPPLIER: EDIT ───────────────────────────────────────────────────── function editSupplier(p) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SUPPLY_SHEET); if (!sheet) return { success: false, error: "ไม่พบ Sheet: " + SUPPLY_SHEET }; const ri = parseInt(p.rowIndex); if (isNaN(ri) || ri < SUPPLY_START_ROW) return { success: false, error: "rowIndex ไม่ถูกต้อง" }; sheet.getRange(ri, 2).setValue(p.name || ""); sheet.getRange(ri, 3).setValue(p.contact || ""); sheet.getRange(ri, 4).setValue(p.phone || ""); sheet.getRange(ri, 5).setValue(p.email || ""); sheet.getRange(ri, 6).setValue(p.category || ""); sheet.getRange(ri, 7).setValue(Number(p.leadtime) || ""); sheet.getRange(ri, 8).setValue(p.note || ""); return { success: true }; } // ─── SUPPLIER: DELETE ─────────────────────────────────────────────────── function delSupplier(p) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SUPPLY_SHEET); if (!sheet) return { success: false, error: "ไม่พบ Sheet: " + SUPPLY_SHEET }; const ri = parseInt(p.rowIndex); if (isNaN(ri) || ri < SUPPLY_START_ROW) return { success: false, error: "rowIndex ไม่ถูกต้อง" }; sheet.deleteRow(ri); renumberSupply(sheet); return { success: true }; } // ─── HELPERS ──────────────────────────────────────────────────────────── function ensureHeaders(sheet) { // เพิ่ม header O5=STARRED, P5=COMMENTS ถ้ายังไม่มี const oVal = sheet.getRange(HEADER_ROW, 15).getValue(); const pVal = sheet.getRange(HEADER_ROW, 16).getValue(); if (!oVal) sheet.getRange(HEADER_ROW, 15).setValue("STARRED"); if (!pVal) sheet.getRange(HEADER_ROW, 16).setValue("COMMENTS"); } function normalizeStatus(raw) { if (raw === true || String(raw).toUpperCase() === "TRUE") return "เสร็จแล้ว ✅"; if (raw === false || String(raw).toUpperCase() === "FALSE") return "กำลังดำเนินการ ♻"; const s = String(raw || "").trim(); return s || "กำลังดำเนินการ ♻"; } function fmtDate(v) { if (!v) return ""; if (v instanceof Date && !isNaN(v)) return Utilities.formatDate(v, "Asia/Bangkok", "d/M/yyyy"); const s = String(v).trim(); return s === "-" ? "" : s; } function fmtNow() { return Utilities.formatDate(new Date(), "Asia/Bangkok", "dd/MM/yyyy HH:mm:ss"); } function updateTimestamp(sheet) { sheet.getRange("A2").setValue("อัปเดตล่าสุด: " + fmtNow()); } function renumberColumn(sheet) { const lastRow = sheet.getLastRow(); if (lastRow < DATA_START_ROW) return; const numRows = lastRow - DATA_START_ROW + 1; const noCol = sheet.getRange(DATA_START_ROW, 1, numRows, 1).getValues(); for (let i = 0; i < noCol.length; i++) { if (noCol[i][0] !== "") noCol[i][0] = i + 1; } sheet.getRange(DATA_START_ROW, 1, numRows, 1).setValues(noCol); } function renumberSupply(sheet) { const lastRow = sheet.getLastRow(); if (lastRow < SUPPLY_START_ROW) return; const numRows = lastRow - SUPPLY_START_ROW + 1; const allCols = sheet.getRange(SUPPLY_START_ROW, 1, numRows, 2).getValues(); let counter = 1; for (let i = 0; i < allCols.length; i++) { if (allCols[i][1]) { allCols[i][0] = counter++; } else { allCols[i][0] = ""; } } sheet.getRange(SUPPLY_START_ROW, 1, numRows, 1).setValues(allCols.map(r => [r[0]])); }
